47 PasswordContext::PasswordContext(
48  const string& password,
49  const string& client,
50  const string& realm,
51  const string& host,
52  const int port
53 ) : Context(client, realm, host, port)
54 {
55  if (password.empty()) {
56  throw bad_pw(KADM5_BAD_PASSWORD);
57  }
58 
59  // FIXME This is buggy in Heimdal. If password is correct, the
60  // the libraries will prompt for the password a second time(!).
61  // If the password was wrong, a bad_pw exception will be thrown
62  // immediately.
63  void* ph = NULL;
64  error::throw_on_error(
65  kadm5_init_with_password_ctx(
66  *this,
67  client.empty() ? NULL : client.c_str(),
68  password.c_str(),
69  KADM5_ADMIN_SERVICE,
70  config_params().get(),
71  KADM5_STRUCT_VERSION,
72  KADM5_API_VERSION_2,
73  &ph
74  )
75  );
76  set_kadm_handle( shared_ptr<void>(ph, kadm5_destroy) );
77 
78  // Check connection.
79  u_int32_t p;
80  error::throw_on_error(
81  kadm5_get_privs(*this, &p)
82  );
83 }
