48 void error::throw_on_error(int32_t c)
49 {
50  switch(c) {
51  case 0L:
52  case KRB5KDC_ERR_NONE:
53  return;
54  case KADM5_FAILURE:
55  throw error(c);
56 
57 
58  // KAdmin general errors
59  case KADM5_BAD_SERVER_HANDLE:
60  throw bad_handle(c);
61  case KADM5_BAD_DB:
62  throw bad_db(c);
63  case KADM5_BAD_HIST_KEY:
64  throw key_history_mismatch(c);
65  case KADM5_SECURE_PRINC_MISSING:
66  throw secure_principal_missing(c);
67  case KADM5_NO_RENAME_SALT:
68  throw salt_prevents_rename(c);
69  case KADM5_BAD_TL_TYPE:
70  throw bad_tl_type(c);
71 
72 
73  // KAdmin config errors
74  case KADM5_BAD_CLIENT_PARAMS:
75  throw remote_config_error(c);
76  case KADM5_BAD_SERVER_PARAMS:
77  throw local_config_error(c);
78  case KADM5_MISSING_CONF_PARAMS:
79  throw params_missing(c);
80  case KADM5_BAD_SERVER_NAME:
81  throw bad_server(c);
82 
83 
84  // KAdmin connection errors
85  case KADM5_RPC_ERROR:
86  throw rpc_error(c);
87  case KADM5_NO_SRV:
88  throw no_server(c);
89  case KADM5_NOT_INIT:
90  throw not_initialized(c);
91  case KADM5_INIT:
92  throw already_initialized(c);
93  case KADM5_BAD_PASSWORD:
94  throw bad_pw(c);
95 
96 
97  // KAdmin authentication errors
98  case KADM5_AUTH_INSUFFICIENT:
99  throw auth_missing(c);
100  case KADM5_AUTH_GET:
101  throw get_auth_missing(c);
102  case KADM5_AUTH_ADD:
103  throw add_auth_missing(c);
104  case KADM5_AUTH_MODIFY:
105  throw modify_auth_missing(c);
106  case KADM5_AUTH_DELETE:
107  throw delete_auth_missing(c);
108  case KADM5_AUTH_LIST:
109  throw list_auth_missing(c);
110  case KADM5_AUTH_CHANGEPW:
111  throw cpw_auth_missing(c);
112 #ifdef KADM5_AUTH_SETKEY
113  case KADM5_AUTH_SETKEY:
114  throw setkey_auth_missing(c);
115 #endif
116 
117 
118  // KAdmin function parameter errors
119  case KADM5_DUP:
120  throw already_exists(c);
121  case KADM5_UNK_PRINC:
122  throw unknown_principal(c);
123  case KADM5_UNK_POLICY:
124  throw unknown_policy(c);
125  case KADM5_BAD_MASK:
126  throw bad_mask(c);
127  case KADM5_BAD_CLASS:
128  throw bad_char_class(c);
129  case KADM5_BAD_LENGTH:
130  throw bad_pw_length(c);
131  case KADM5_BAD_POLICY:
132  throw bad_policy(c);
133  case KADM5_BAD_PRINCIPAL:
134  throw bad_principal(c);
135  case KADM5_BAD_AUX_ATTR:
136  throw bad_aux_attr(c);
137  case KADM5_BAD_HISTORY:
138  throw bad_pw_history(c);
139  case KADM5_BAD_MIN_PASS_LIFE:
140  throw bad_min_pw_life(c);
141  case KADM5_POLICY_REF:
142  throw policy_in_use(c);
143  case KADM5_PROTECT_PRINCIPAL:
144  throw principal_protected(c);
145 #ifdef KADM5_SETKEY_DUP_ENCTYPES
146  case KADM5_SETKEY_DUP_ENCTYPES:
147  throw duplicate_enctype(c);
148 #endif
149 
150 
151  // KAdmin password quality errors
152  case KADM5_PASS_Q_TOOSHORT:
153  throw pw_too_short(c);
154  case KADM5_PASS_Q_CLASS:
155  throw too_few_char_classes(c);
156  case KADM5_PASS_Q_DICT:
157  throw pw_in_dictionary(c);
158  case KADM5_PASS_REUSE:
159  throw pw_reuse(c);
160  case KADM5_PASS_TOOSOON:
161  throw too_soon(c);
162 
163 
164  // KAdmin version errors
165  case KADM5_BAD_STRUCT_VERSION:
166  throw bad_struct_version(c);
167  case KADM5_OLD_STRUCT_VERSION:
168  throw old_struct_version(c);
169  case KADM5_NEW_STRUCT_VERSION:
170  throw new_struct_version(c);
171  case KADM5_BAD_API_VERSION:
172  throw bad_api_version(c);
173  case KADM5_OLD_LIB_API_VERSION:
174  throw old_lib_api(c);
175  case KADM5_OLD_SERVER_API_VERSION:
176  throw old_server_api(c);
177  case KADM5_NEW_LIB_API_VERSION:
178  throw new_lib_api(c);
179  case KADM5_NEW_SERVER_API_VERSION:
180  throw new_server_api(c);
181 
182 
183  // Kerberos 5 and system errors
184  // (actually returned values only --- see Kerberos 5 API for
185  // details on which function may return which value).
186 
187  // Treat malformed principal name errors like
188  // KADM5_BAD_PRINCIPAL
189  case KRB5_PARSE_MALFORMED:
190  throw bad_principal(c);
191  case ENOMEM:
192  // FIXME Have a preallocated instance to throw on
193  // memory shortage. How does this affect thread-
194  // safety?
195  throw std::bad_alloc();
196  case EINVAL:
197  throw bad_param(c);
198 
199  default:
200  throw error(c);
201  }
202 }
